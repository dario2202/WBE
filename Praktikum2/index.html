Exercises of praktika 2

Prime exercise

    
        function isPrime (n){
            for (let index = 2; index < n; index++) {
               if(n % index === 0) return false; //Return false if Modul 0
           }
           return true;
        }
        
        
        function isPrime2 (n){
            for (let index = 2; index < n; index++) {
               if(n % index === 0) return 0; //Return false if Modul 0
           }
           return 1;
        }

Ist es wichtig ob true/false oder 1/0 zurückgegeben wird?

Die beiden Funktionen geben einen unterschiedlichen Wert zurück. Da aber eine implizite Typenkonvertierung passiert geben sie dennoch den gleichen Wert zurück. Hier ist ein Beispiel zu sehen: 

    
        document.write("Output:")
        document.write(isPrime(19)==isPrime2(19))

Fibonacci Exercise (Exercise 4)

Subexercise a

        
            function fibo (n) {
                if(n === 0){
                    return 0
                }else if(n === 1){
                    return 1
                }else{
                    return fibo(n-1) + fibo(n-2)
                }
            }

Subexercise b

        
            const g = (1+ Math.sqrt(5))/2 //Goldener Schnitt
            function fiboAdvanced(n){
                return Math.round(Math.pow(g,n) / Math.sqrt(5))
            }

Subexercise d

        
            function fibonacciBigInt(n){
                let summe = 0;
                let erste = BigInt(0);
                let zweite = BigInt(1);

                for (let i = BigInt(2); i <= n; i += BigInt(1)) {
                    summe = erste + zweite;
                    erste = zweite;
                    zweite = summe;
                }

                return summe;
            }

            console.log("fib(1000) = " + fibonacciBigInt(1000));
